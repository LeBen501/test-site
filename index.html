import json import js async def find_athlete(): # Eingaben holen height = int(Element("height").element.value) weight = int(Element("weight").element.value) sex = Element("sex").element.value # JSON-Datei laden response = await js.fetch("athletes.json") data = await response.json() # Daten filtern matching_athletes = [a for a in data if a["Height"] == height and a["Weight"] == weight and a["Sex"] == sex] # Bester Athlet ermitteln if not matching_athletes: Element("result").element.innerText = "Kein passender Athlet gefunden." else: best_athlete = max(matching_athletes, key=lambda x: x["Medals"]) Element("result").element.innerText = f"Erfolgreichster Athlet: {best_athlete['Name']} mit {best_athlete['Medals']} Medaillen!"
